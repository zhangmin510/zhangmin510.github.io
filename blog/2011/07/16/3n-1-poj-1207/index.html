    int calCycleLength(int n){
    int length = 1;
    while(1 != n){
    length ++;
    if(n % 2 == 0){
    n = n/2;
    }
    else{
    n = 3    n + 1;
    }
    }
    return length;
    }

    max = 0;
    if(i< j){//i may be less than j
    begin = i;
    end = j;
    }else{
    begin = j;
    end = i;
    }
    for(n = begin;n <= end;n++){ tmp = calCycleLength(n); if(tmp > max){
    max = tmp;
